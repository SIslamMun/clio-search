\newcommand\setfootervars[1]{
  \setsepchar{,}
  \readlist*\params{#1}
  \foreachitem\param\in\params{
    \setsepchar{=}
    \readlist*\kv{\param}
    \expandafter\edef\csname footer\kv[1]\endcsname{\kv[2]}
  }
}

\newcommand\footerwithtwologos[1]{
  \setfootervars{#1}
  \input{includes/footer/two-logos}
}

\newcommand\footerwiththreelogos[1]{
  \setfootervars{#1}
  \input{includes/footer/three-logos}
}

\newcommand\footerwithlogos[1]{
  \setfootervars{#1}
  \input{includes/footer/dynamic}
}
